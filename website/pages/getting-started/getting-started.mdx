## Overview

Chakra UI Steps is a React component which has been designed to integrate smoothly with <a href="https://chakra-ui.com" rel="no_referrer">chakra-ui</a>.

## Installation

`npm install chakra-ui-steps`
or using yarn
`yarn add chakra-ui-steps`

## Configure Theme

In order to start using the component, you need to extend the default theme which ships with Chakra UI. To do so

```tsx
import { chunk } from "lodash";
import * as React from "react";
import { Col, Row, Grid } from "react-flexbox-grid";

type GridGeneratorProps = {
  cols: 1 | 2 | 3 | 4 | 6 | 12;
};

const GridGenerator = ({ cols, children }: GridGeneratorProps) => {
  const colWidth = 12 / cols;
  const rows = chunk(React.Children.toArray(children), cols);
  return (
    <Grid>
      {rows.map((cols) => (
        <Row>
          {cols.map((col) => (
            <Col sm={12} md={colWidth}>
              {col}
            </Col>
          ))}
        </Row>
      ))}
    </Grid>
  );
};

export default GridGenerator;
```
